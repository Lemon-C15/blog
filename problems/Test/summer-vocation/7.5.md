## H.steaks
* https://codeforces.com/gym/102263/problem/H
* 切牛排，一个锅可以放两个，有n个牛排，煎一面5分钟，问最少几分钟
* n=3，k=1；a1+b1..a2+c1..b2+c2,最少15分钟
* 思路：肉分两面，锅分两瓣，一个锅链接两块不同的肉，此时一次最高效率，不能刚好加一次，总能链接到不同的肉
 
* 1. 
```c++
        #include <bits/stdc++.h>

            }

            if(N) N=5;
            else N=0;
            n=n/k;
            long long int time;
            time=n*5;
            cout<<time+N;
            return 0;
        }
```

## B.博弈 road to arabella
* https://vjudge.net/problem/Gym-102263B
* 先给个T，有T回合，给n和k，每次取x={1<=x<=max(1,m-k)},m为此时的n，然后使n=n-x，谁先让n等于0，让对方没得动就赢，a先手
* 思路：给对方偶数必赢，当m<=k或m-k=1，则只能减1，若此时a是偶数，减下去a必赢，若此时m>k,中间仍然可以只减一次一个特殊数使给到b的数为偶且以后只能减1，类似test2.
* test1：2 1；此时只能减1，b先减到0，b赢
* test2: 4 1；此时可以减1到3，减2，给b偶数2使b只能减1，a赢
* a先，若此时可以不只减1，则a必赢，若只能减1，a是奇数a必赢，否则b赢
```c++
#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	while(n--)
	{
		int a,b;
		cin>>a>>b;
		if(a==b||a-1==b)// 0 or 1
		{
			if(a&1)printf("Kilani\n");//奇偶，a为奇数返回1
			else printf("Ayoub\n");
		}
		else printf("Kilani\n");
	}
 
	return 0;
}
```

## M 消消乐
* https://vjudge.net/problem/Gym-102263M
* 给个字符串，使他从x排到a，若有两个一样则替换成把这两个替换成下一个字母，bb=c
```c++
#include<bits/stdc++.h>
using namespace std;
string s;
int arr[26];
int main(){
    ios::sync_with_stdio;
    cin>>s;
    memset(arr,26,0);
    for(int i=0;i<s.length();i++){
        arr[s[i]-'a']++;
    }
    for (int i = 0; i < 25; i++)//正推
    {
        if(arr[i]==0) continue;
        arr[i + 1] +=arr[i]/2;// 1/2=0
        if (arr[i] % 2 != 0) arr[i] = 1;
        else arr[i]=0;
    }
    for(int i=0;i<arr[25];i++){
        cout<<'z';
    }
    for(int i=24;i>=0;i--){
        if(arr[i]==0) continue;
        else cout<<char(i+'a');
    }
    cout<<endl;
    return 0;
}

```


## C 文本匹配
* https://vjudge.net/problem/Gym-102263C
* 给一个文本，输入'Space','Backspace','CapsLock',和小写字母，看结果和文本相同否，同则Correct，否则Incorrect
* 用队列表示输入字符串，然后匹配即可
* 1.
```c++
#include<bits/stdc++.h>
using namespace std;
queue<string> qq;//队列
deque<char> st;//迭代器
int n,m;
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        string temp;
        cin>>temp;
        qq.push(temp);//整条存储
    }
    scanf("%d",&m);//操作数
    bool cc=false;//大小写切换
    for(int i=0;i<m;i++){
        string temp;//归零
        cin>>temp;
        if(temp.length()>1){//输入操作
            if(temp[0]=='B'){
                if (!st.empty())//不能无法退格
                    st.pop_back();
            }
            else if(temp[0]=='C'){
                cc=!cc;//大小写切换
            }
            else{
                st.push_back(' ');//空格
            }
        }
        else{//输入字母
            char c=temp[0];
            if(cc) c=c-32;//大小写判断，cc=1为大写，-32
            st.push_back(c);
        }
    }
    string com="";//开新的两条存储对比项目
    string org="";
    while(!qq.empty()){//empty表示队列空返回1，此时表示若 非空 则。。
        org+=qq.front();//arr.front();返回的是第一个元素的引用。arr.back();返回的的是最后一个元素的引用。
        qq.pop();//一条一条拿，一条一条删除
        if(!qq.empty())
            org+=' ';
    }
    while (!st.empty())
    {
        com+=st.front();
        st.pop_front();
    }
    bool res=true;
    if(org.size()!=com.size()) res=false;
    else{
        for(int k=0;k<org.size();k++){
            if(org[k]!=com[k]){
                res=false;
                break;
            }
        }
    }
    if (res)
        cout << "Correct" << endl;
    else
        cout << "Incorrect" << endl;
    return 0;
}
```
