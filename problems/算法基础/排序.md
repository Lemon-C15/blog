# 快排
```c++
void sort1(int l,int r){//快排
    if(l>=r) return;
    int i=l,j=r;
    int x = a[l];
    while(i != j)
    {
        while(i < j && a[j] <=x) //改变符号即可降序
            j--;     
        while(i < j && a[i]>= x)
            i++;
        int t;
        t=a[i];
        a[i] = a[j];
        a[j] = t;
    }
    a[l]=a[i];
    a[i]=x;   
    sort1(l, i - 1);
    sort1(i + 1, r);
}

```
# 二分
```c++
int Binary_search(int array[], int l, int r, int x){
    int mid=(l+r)/2;
    while(l<=r){
        if(array[mid]==x) return mid;
        if(x<array[mid]) r=mid-1;
        else l=mid+1;
        mid=(l+r)/2;//****
    }
    return -1;
}
```
