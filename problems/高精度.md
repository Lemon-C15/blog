# High-add
* 关键点：
* * 边界i<a.size()||i<b.size

```c++
t=t+A[i]+b[i];
```

```c++
#include <iostream>
#include <vector>

using namespace std;
const int N = 1e6+10;

vector<int> add(vector<int> A,vector<int> B){
    vector<int> c;
    int t=0;
    for(int i=0;i<A.size()||i<B.size();i++){
        if(i<A.size()) t+=A[i];
        if(i<B.size()) t+=B[i];
        c.push_back(t%10);
        t=t/10;
    }
    return c;
}

int main(){
    string a,b;
    vector<int> A,B;
    cin>>a>>b;
    for(int i=a.size()-1;i>=0;i--) A.push_back(a[i]-'0');
    for(int i=b.size()-1;i>=0;i--) B.push_back(b[i]-'0');
    auto c=add(A,B);
    for(int i=c.size()-1;i>=0;i--) printf("%d",c[i]);
    return 0;
}
```

# High-sub
* 关键点
* * 计算t可以用c.push_back((t+10)%10),无论怎样都是正
* * 永远大减小，输出时添加负号



```c++
#include <iostream>
#include <cstring>
#include <vector>
#define N 1e6+10
#define ll long long int
using namespace std;

vector<int> sub(vector<int> A,vector<int> B){
    vector<int> c;
    int t=0;
    for(int i=0;i<A.size();i++){
        t+=A[i];
        if(i<B.size()) t-=B[i];
         
        if(t>=0){
            c.push_back(t);
            t=0;
        }
        else if(t<0){
            
            c.push_back(10+t);
            t=-1;      
            
        }   
    }
    
    return c;
}

int main(){
    string a,b;
    cin>>a>>b;
    vector<int> A,B,c;
    for(int i=a.size()-1;i>=0;i--) A.push_back(a[i]-'0');
    for(int i=b.size()-1;i>=0;i--) B.push_back(b[i]-'0');
    if(A.size()==B.size()){
        if(a>b) c=sub(A,B);
        else if(a<b){
            c=sub(B,A);
            printf("-");
        }
        else {}
    }
    else if(A.size()>B.size()){
        c=sub(A,B);
        
    }       
    else{
        c=sub(B,A);
        printf("-");    
    }

    while(c.size()>1&&c.back()==0) c.pop_back();
    if(a==b) printf("0");
    else{
        for(int i=c.size()-1;i>=0;i--){
            printf("%d",c[i]);
        }
    }
    
    return 0;
}
```
