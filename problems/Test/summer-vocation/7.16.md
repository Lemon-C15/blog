## D - Husam's Bug
* 题目概括：
* * 一个字符串应该包含4个字母4个数字和两个符号，可重复，字母大写小写都可以，符号从“@，？，！”中选
* * 字符串最短为9，最长为100
* 缺少什么则输出对应结果
* 思路：穷举
```c++
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin>>n;
    for(int I=0;I<n;I++){
        int ans[5];
        memset(ans,0,sizeof(ans));
        string s;
        cin>>s;
        for(int i=0;i<s.size();i++){    
            if(s[i]=='!'||s[i]=='@'||s[i]=='?') ans[1]++;
            else if(('a'<=s[i]&&s[i]<='z')||('A'<=s[i]&&s[i]<='Z')) ans[2]++;
            else if('0'<=s[i]&&s[i]<='9') ans[3]++;
        }
        if(ans[1]<2) cout<<"The last character must be a symbol.";
        else if(ans[2]<4) cout<<"The last character must be a letter.";
        else if(ans[3]<4) cout<<"The last character must be a digit.";
        else cout<<"The last character can be any type.";
        cout<<endl;
    }
    return 0;
}
```
## F - Certifications
* 题目概括:n个数字，m次询问，每次输出大于等于输入的，没有就输出':(';边界1e9
* 思路：数据大，用二分
```c++
#include <bits/stdc++.h>
using namespace std;
const int N=1e5+10;
int a[N];

int find(int x,int n){
    int l=0,r=n,mid;
    mid=(l+r)/2;
    while(l<=r){
        if(a[mid]<x) l=mid+1;//不可等，因为要返回大于等于那个
        else r=mid-1;
        mid=(l+r)/2;
    }
    return l;
}

int main(){
    int n,m;
    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];
    sort(a,a+n);
    cin>>m;
    for(int i=0;i<m;i++){
        int x;
        cin>>x;
        int j=find(x,n-1);
        if(j==n) cout<<"Dr. Samer cannot take any offer :(." <<endl;
        else cout<<a[j]<<endl;
    }
    return 0;
}
```
## E - Abdalrahman Ali Bugs
### 题目概括
* 给一条字符串和公式，代入a^i，a^i是第i个字母的出现次数，可重复，求使公式和最小的参数x ；
* 长度1e5,全小写    
### 样例
```C++
acm 2
aababaa 5
abcabcabc   3
```
### 思路
* 看起来是出现最多的字母数，实际上不是
* 先求出每个位置的a^i,因为数据小，所以穷举
```c++
#include <bits/stdc++.h>
using namespace std;
 
const int maxn = 1e5 + 10;
int n, m;
char s[maxn];
long long f[maxn];
 
int main() {
  scanf("%s", s);
  for(int i = 0; s[i]; i ++) {
    f[s[i] - 'a'] ++;
  }
  long long ans = -1;
  long long mn = -1;
  for(long long i = 2; i <= 300000; i ++) {
    long long tmp = 0;
    for(int j = 0; j < 26; j ++) {
      tmp = tmp + (f[j] % i) * f[j];
    }
    if(mn == -1 || tmp < mn) {
      mn = tmp;
      ans = i;
    }
  }
  printf("%lld\n", ans);
  return 0;
}
```
## H - Give Me This Pizza
* 题目概括：输出每一个数右边第一个大于自身的数，否则-1；
* 思路：单调栈，栈里递减，没有重复
```c++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e5+10;
int s[N],a[N],idx=0;
int main(){
    int n;
    cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i];
    s[0]=-1;
    for(int i=n;i>0;i--){
        while(a[i]>=s[idx]&&idx>0) idx--;//没有相同的
        int t=a[i];
        a[i]=s[idx];
        idx++;
        s[idx]=t;
    }
    for(int i=1;i<=n;i++) cout<<a[i]<<" ";
    return 0;
}
```
## C - MRT Map
* 题目概括：给n个城市，两两城市之间的路线长度是相同字母（不区分大小写，不记数量）的数量，然后求最短路的路程
* 思路：用dijkstra
```c
#include <bits/stdc++.h>
#define mm(a) memset(a, 0, sizeof(a))
using namespace std;
typedef long long ll;
#define LLu unsigned long long
#define PII pair<ll, ll>
#define xx first
#define yy second
ll gcd(ll a, ll b) {return b ? gcd(b, a%b) : a; }
const int INF = 0x3f3f3f3f;
const int N = 1e5 + 7;
char a[N][33];
int n, m;
vector<int>mp[N];
queue<int>q;
int dis[N];
int vis[N];
int cha(int x, int y)
{
    int ans = 0;
    int c1[444], c2[444];
    mm(c1);
    mm(c2);
    int n1 = strlen(a[x]), n2 = strlen(a[y]);
    for(int i = 0; i < n1; i ++)
    {
        if(a[x][i]>='A'&&a[x][i]<='Z') a[x][i] += 32;
        c1[a[x][i]] ++;
    }
    for(int i = 0; i < n2; i ++)
    {
        if(a[y][i]>='A'&&a[y][i]<='Z') a[y][i] += 32;
        c2[a[y][i]] ++;
    }
    for(char i = 'a'; i <= 'z'; i ++)
    {
        if(c1[i] && c2[i]) ans ++;
    }
    return ans;
}
void dijkstra()
{
    while(!q.empty())
    {
        int now, no;
        now = q.front();
        q.pop();
        vis[now] = 0;
        for(int i = 0; i < mp[now].size(); i ++)
        {
            int w = cha(now, mp[now][i]);
            if(dis[mp[now][i]] > dis[now] + w)
            {
 
                dis[mp[now][i]] = dis[now] + w;
                if(!vis[mp[now][i]])
                {
                    vis[mp[now][i]] = 1;
                    q.push(mp[now][i]);
                }
 
            }
        }
    }
}
int main()
{
    cin >> n >> m;
    for(int i = 1; i <= n; i ++)
    {
        cin >> a[i];
    }
    memset(dis, 0x3f, sizeof(dis));
    memset(vis, 0, sizeof(vis));
    while(m --)
    {
        int u, v;
        cin >> u >> v;
        mp[u].push_back(v);
        mp[v].push_back(u);
    }
    int s, e;
    cin >> s >> e;
    while(!q.empty()) q.pop();
    q.push(s);
    vis[s] = 1;
    dis[s] = 0;
    dijkstra();
    cout << dis[e] << endl;
    return 0;
}
```
